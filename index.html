 PRINT("HALLO")

 print("hey")
END

Schleifen

--RAUF
FOR I = 1,10 DO
 PRINT(I)
END

--RUNTER
FOR A=10,0,-2 DO
 PRINT(A)
END

--ARRAY
FOR S IN ALL(SHIPS) DO
 PRINT(S.NAME)
END

FOR K,V IN PAIRS(M) DO
 PRINT("K:"..K..",V:"..V)
END
